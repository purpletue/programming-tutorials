<!-- 2) Basic document structure -->
<!-- ---------------------------------------------------------------- -->
<!-- The basic skeleton looks like this:
     <!DOCTYPE html>
     <html lang="en">
       <head>
         <meta charset="UTF-8">
         <title>Page Title</title>
       </head>
       <body>
         <!-- content -->
       </body>
     </html>
 -->
<!-- Learn the purpose of the doctype, html, head, title, and body elements. -->
<!-- ---------------------------------------------------------------- -->
<!-- 3) Text and inline elements -->
<!-- ---------------------------------------------------------------- -->
<!-- Use headings, paragraphs, links, emphasis, and lists.
     Examples:
     <h1>Heading</h1>
     <p>This is a paragraph.</p>
     <a href="https://example.com">Link</a>
     <strong>Strong text</strong>
     <em>Emphasized text</em>
     <ul><li>Item</li></ul>
 -->
<!-- Learn when to use block-level vs inline elements and semantic tags. -->
<!-- ---------------------------------------------------------------- -->
<!-- 4) Images, media, and assets -->
<!-- ---------------------------------------------------------------- -->
<!-- Use <img> for images and <audio>/<video> for media.

 -->
